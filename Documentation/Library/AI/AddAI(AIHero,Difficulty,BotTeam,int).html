Deltin.CustomGameAutomation.AI.AddAI(AIHero hero, Difficulty difficulty, BotTeam team, [int count]) Method
Add AI to the game.
ParameterTypeDescription
hero
AIHeroHero type to add.
difficulty
DifficultyDifficulty of hero.
team
BotTeamTeam that AI joins.
count
intAmount of AI that is added. Set to -1 for max. Default is -1
Returns
Type: bool
Returns false if no AI can be added.
ExceptionThrown
ArgumentOutOfRangeExceptionThrown if count is less than -1.
Examples
The code below will even out the teams with AI if the teams are unbalanced. 
using Deltin.CustomGameAutomation;
using System;
using System.Linq;

public class AddAIExample
{
    const int PlayerCountDifference = 1;

    public static void EvenPlayerTeams(CustomGame cg)
    {
        cg.AI.CalibrateAIChecking();

        int dif = cg.BlueCount - cg.RedCount; // Will be positive if blue has more players and negative if red has more players.

        if (Math.Abs(dif) < PlayerCountDifference)
            return;

        if (dif < 0) // Red has more players  
        {
            // Remove extra AI from red
            int[] redAI = cg.AI.GetAISlots().Where(aiSlot => cg.IsSlotRed(aiSlot)).ToArray();
            for (int i = 0; i < redAI.Length && i < -dif; i++)
                if (cg.AI.RemoveFromGameIfAI(redAI[i]))
                    dif++;

            if (dif != 0)
                cg.AI.AddAI(AIHero.Recommended, Difficulty.Hard, BotTeam.Blue, -dif);
        }
        else if (dif > 0) // Blue has more players
        {
            // Remove extra AI from blue
            int[] blueAI = cg.AI.GetAISlots().Where(aiSlot => cg.IsSlotBlue(aiSlot)).ToArray();
            for (int i = 0; i < blueAI.Length && i < dif; i++)
                if (cg.AI.RemoveFromGameIfAI(blueAI[i]))
                    dif--;

            if (dif != 0)
                cg.AI.AddAI(AIHero.Recommended, Difficulty.Hard, BotTeam.Red, dif);
        }
    }
}
