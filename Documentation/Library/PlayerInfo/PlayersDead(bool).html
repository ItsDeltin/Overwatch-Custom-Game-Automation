Deltin.CustomGameAutomation.PlayerInfo.PlayersDead([bool noUpdate]) Method
Gets a list of players who died.
ParameterTypeDescription
noUpdate
bool
Returns
Type: List<int>
List of players who are dead.
Examples
The code below will award a point to a team if they get a kill. 
using System;
using System.Collections.Generic;
using System.Threading;
using Deltin.CustomGameAutomation;
            
public class PlayersDeadExample
{
    public static void AwardTeamOnKill(CustomGame cg)
    {
        bool[] awarded = new bool[12];
        
        int redpoints = 0;
        int bluepoints = 0;
        
        while (true)
        {
            List<int> playersdead = cg.PlayerInfo.PlayersDead();
            for (int i = 0; i < 12; i++)
            {
                if (playersdead.Contains(i) && awarded[i] == false)
                {
                    if (cg.IsSlotBlue(i))
                    {
                        redpoints++;
                        Console.WriteLine("Red points: {0}", redpoints);
                    }
                    else if (cg.IsSlotRed(i))
                    {
                        bluepoints++;
                        Console.WriteLine("Blue points: {0}", bluepoints);
                    }
                    awarded[i] = true;
                }
                else if (playersdead.Contains(i) == false && awarded[i])
                {
                    awarded[i] = false;
                }
            }
            Thread.Sleep(500);
        }
    }
}
