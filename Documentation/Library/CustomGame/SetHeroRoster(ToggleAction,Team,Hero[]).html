Deltin.CustomGameAutomation.CustomGame.SetHeroRoster(ToggleAction ta, Team team, Hero[] heroes) Method
Toggles what heroes can be selected.
ParameterTypeDescription
ta
ToggleActionDetermines if all heroes should be enabled, disabled or neither before toggling
team
TeamTeam to change roster for.
heroes
Hero[]Heroes to toggle.
ExceptionThrown
ArgumentOutOfRangeExceptionThrown if team is Spectator or Queue.
Examples
The example below will enable random heroes for a team. 
using System;
using System.Collections.Generic;
using System.Linq;
using Deltin.CustomGameAutomation;

public class SetHeroRosterExample
{
    public static void ChooseRandomHeroes(CustomGame cg, Team team, int randomHeroCount)
    {
        Random rnd = new Random();
        int heroCount = Enum.GetNames(typeof(Hero)).Length - 1;

        List<Hero> chooseHeroes = new List<Hero>();
        while (chooseHeroes.Count < randomHeroCount)
        {
            int heroID = rnd.Next(heroCount);
            if (chooseHeroes.Select(hero => (int)hero).Contains(heroID))
                continue;
            chooseHeroes.Add((Hero)heroID);
        }

        cg.SetHeroRoster(ToggleAction.DisableAll, team, chooseHeroes.ToArray());
    }
}
