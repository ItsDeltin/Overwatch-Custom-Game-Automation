public void Chat(
    string text

text
System.String
String to send to chat ingame. CustomGame has a string called ChatPrefix which adds a prefix to your chat messages.

Sends a message to the chat ingame.

The example below will send a welcome message every time a player joins.

using System;
using System.Threading;
using Deltin.CustomGameAutomation;

...

static void Main(string[] args)
{
    CustomGame cg = new CustomGame();

    int preplayercount = 0;

    while (true)
    {
        int playercount = cg.PlayerCount;
        if (playercount > preplayercount)
        {
            cg.Chat.JoinChannel(Channel.Match);
            cg.Chat.Chat(string.Format("Welcome to the game! There are currently {0}/{1} players in this game.", playercount, cg.PlayerInfo.MaxPlayerCount()[0] + cg.PlayerInfo.MaxPlayerCount()[1]));
        }
        preplayercount = playercount;
        Thread.Sleep(50);
    }
}
