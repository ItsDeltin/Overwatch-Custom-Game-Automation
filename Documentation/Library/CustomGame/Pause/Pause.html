CustomGame.CG_Pause.Pause()
Syntax 
public void Pause()

Pauses the server if the server is unpaused, else it does nothing.

using System;
using Deltin.CustomGameAutomation;

...

static void Main(string[] args)
{
    // 5 second countdown for race.

    CustomGame cg = new CustomGame();

    cg.Pause.Pause(); // Pause game.
    cg.Pause.Unpause(); // Unpause and start 5 second countdown.
    cg.Chat.Chat("Get ready to race!");
}
