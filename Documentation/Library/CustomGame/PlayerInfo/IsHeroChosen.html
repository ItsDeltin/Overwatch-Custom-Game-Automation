Returns true if the player chose a hero.

Checks if targetted player has chosen a hero.

The code below will switch everyone who hasn't chosen a hero to the opposite team.

using System;
using System.Collections.Generic;
using Deltin.CustomGameAutomation;

...

static void Main(string[] args)
{
    CustomGame cg = new CustomGame();

    List<int> slots = cg.PlayerSlot;

    // Check each slot filled
    for (int i = 0; i < slots.Length; i++)
    {
        // Check if player chose a hero
        if (cg.PlayerInfo.IsHeroChosen(slots[i]) == false) 
        {
            // If not, switch them to the opposite team.
            cg.Interact.SwapTeam(slots[i]);
        }
    }
}
